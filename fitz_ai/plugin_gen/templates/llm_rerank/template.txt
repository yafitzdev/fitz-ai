# TEMPLATE: LLM Rerank Plugin
#
# Rerank plugins re-order retrieved chunks by relevance to a query.
#
# REQUIRED FIELDS:
#   - plugin_name: Unique identifier
#   - plugin_type: Must be "rerank"
#   - provider.base_url: API base URL
#   - auth.env_vars: Environment variable(s) for API key
#   - endpoint.path: API endpoint path
#   - defaults.model: Default reranking model name
#   - request.query_key: Key name for query in request
#   - request.documents_key: Key name for documents array in request
#   - response.results_path: JSONPath to extract reranked results

plugin_name: "CHANGE_ME"
plugin_type: "rerank"
version: "1.0"

provider:
  name: "CHANGE_ME"
  base_url: "https://api.example.com/v1"  # API base URL

auth:
  type: "bearer"  # "bearer" | "header" | "none"
  header_name: "Authorization"
  header_format: "Bearer {key}"
  env_vars:
    - "EXAMPLE_API_KEY"  # Environment variable for API key

endpoint:
  path: "/rerank"  # API endpoint path
  method: "POST"
  timeout: 60

defaults:
  model: "rerank-model"  # Reranking model name
  top_n: 10  # Number of results to return

request:
  query_key: "query"  # Key for query in request body
  documents_key: "documents"  # Key for documents array
  param_map:
    model: "model"
    top_n: "top_n"

response:
  # JSONPath to extract reranked results from API response
  results_path: "results"
  # Each result expected to have: index, relevance_score
