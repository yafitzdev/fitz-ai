# TEMPLATE: LLM Embedding Plugin
#
# Embedding plugins convert text to vector representations.
#
# REQUIRED FIELDS:
#   - plugin_name: Unique identifier
#   - plugin_type: Must be "embedding"
#   - provider.base_url: API base URL
#   - auth.env_vars: Environment variable(s) for API key
#   - endpoint.path: API endpoint path
#   - defaults.model: Default embedding model name
#   - defaults.dimensions: Vector dimensions (e.g., 768, 1536)
#   - request.input_key: Key name for input text in request
#   - response.embedding_path: JSONPath to extract embedding vector

plugin_name: "CHANGE_ME"
plugin_type: "embedding"
version: "1.0"

provider:
  name: "CHANGE_ME"
  base_url: "https://api.example.com/v1"  # API base URL

auth:
  type: "bearer"  # "bearer" | "header" | "none"
  header_name: "Authorization"
  header_format: "Bearer {key}"
  env_vars:
    - "EXAMPLE_API_KEY"  # Environment variable for API key

endpoint:
  path: "/embeddings"  # API endpoint path
  method: "POST"
  timeout: 60

defaults:
  model: "text-embedding-model"  # Embedding model name
  dimensions: 1536  # Output vector dimensions

request:
  input_key: "input"  # Key for input text in request body
  param_map:
    model: "model"

response:
  # JSONPath to extract embedding vector from API response
  embedding_path: "data[0].embedding"
